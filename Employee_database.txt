1)FETCH THE EMPOLYEE NAME AND DEPARTMENT NAME(INNER JOIN)
SELECT E.ENAME,D.DNAME FROM EMPLOYEE E JOIN DEPARTMENT D WHERE E.D_ID=D.D_ID

2)FETCH ALL THE EMPLOYEE NAME AND THERE DEPARTMENT NAME(LEFT JOIN)
SELECT E.ENAME,D.DNAME FROM EMPLOYEE E LEFT JOIN DEPARTMENT D WHERE E.D_ID=D.D_ID

3)FETCH THE EMPLOYEE NAME AND ALL THERE DEPARTMENT NAME(RIGHT JOIN)
SELECT E.ENAME,D.DNAME FROM EMPLOYEE E RIGHT JOIN DEPARTMENT D WHERE E.D_ID=D.D_ID

4)FETCH ALL THE EMPOLYEE NAME AND ALL DEPARTMENT NAME(FULL OUTER JOIN)
SELECT E.ENAME,D.DNAME FROM EMPLOYEE E FULL OUTER JOIN DEPARTMENT D WHERE E.D_ID=D.D_ID
